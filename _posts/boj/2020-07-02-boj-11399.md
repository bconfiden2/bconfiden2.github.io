- 각각을 페어로 만들어서 저장하고(맵을 쓰든 벡터를 쓰든) 시간순에 따라 정렬하면 될 것 같다. -> 생각해보니 굳이 페어로 만들 것도 없이 시간만 저장해서 구하면 될 듯
- 이것도 왜 실버냐...

---
<br>

{% highlight c++ %}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(void)
{
    int n;
    cin >> n;
    vector<int> v(n, 0);
    for(int i = 0 ; i < n ; i++)
    {
        cin >> v[i];
    }
    sort(v.begin(), v.end());
    int ans = 0, temp = 0;
    for(int i = 0 ; i < n ; i++)
    {
        // 해당 순서의 사람이 걸리는 시간
        temp += v[i];
        // 그 사람이 걸린 시간을 전체합에 더해줌
        ans += temp;
    }
    cout << ans;
}
